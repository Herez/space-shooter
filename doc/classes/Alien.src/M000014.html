  def update
    # move
    @x += @speedx
    @y += @speedy
    
    # destroy when the alien reaches the bottom of the screen
    kill! if @y > Engine::Game::ScreenHeight + @image[0].height / 2
    
    # animation
    @index = Gosu::milliseconds / AnimTime % @image.size
  end
